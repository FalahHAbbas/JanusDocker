205 #define JANUS_VALIDATE_JSON_OBJECT_FORMAT(missing_format, invalid_format, obj, params, error_code, error_cause, log_error, missing_code, invalid_code) \
206  do { \
207  error_code = 0; \
208  unsigned int i; \
209  for(i = 0; i < sizeof(params) / sizeof(struct janus_json_parameter); i++) { \
210  json_t *val = json_object_get(obj, params[i].name); \
211  if(!val) { \
212  if((params[i].flags & JANUS_JSON_PARAM_REQUIRED) != 0) { \
213  error_code = (missing_code); \
214  if(log_error) \
215  JANUS_LOG(LOG_ERR, missing_format "\n", params[i].name); \
216  if(error_cause != NULL) \
217  g_snprintf(error_cause, sizeof(error_cause), missing_format, params[i].name); \
218  break; \
219  } \
220  continue; \
221  } \
222  if(!janus_json_is_valid(val, params[i].jtype, params[i].flags)) { \
223  error_code = (invalid_code); \

225  janus_get_json_type_name(params[i].jtype, params[i].flags, type_name); \
226  if(log_error) \
227  JANUS_LOG(LOG_ERR, invalid_format "\n", params[i].name, type_name); \
228  if(error_cause != NULL) \
229  g_snprintf(error_cause, sizeof(error_cause), invalid_format, params[i].name, type_name); \
230  break; \
231  } \
232  } \
233  } while(0)
